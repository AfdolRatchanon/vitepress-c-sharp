# เฉลย Lab: Grade Checker Program

## 📝 เฉลยฉบับสมบูรณ์ (ครอบคลุมทั้งแบบฝึกหัดพื้นฐานและภารกิจท้าทายทั้ง 3 ภารกิจ)

นี่คือ "โค้ดในระดับมืออาชีพ" (Production-ready Code) ที่ไม่เพียงแต่ทำงานตามเงื่อนไขพื้นฐานได้ แต่ยังรวมเอาการป้องกันโปรแกรมพังด้วย `TryParse`, การใช้ Switch Expression เพื่อหา GPA, และการใช้ Ternary Operator เพื่อให้คำแนะนำในตอนท้าย!

กรุณาใช้เวลาศึกษาโครงสร้างของโค้ดนี้อย่างละเอียด เพราะมันรวบรวมแก่นแท้ทั้งหมดของสิ่งที่ C# นำเสนอในบทเรียนที่ผ่านมา

```csharp
using System; // ต้องมีการ using System สำหรับใช้ Console และฟังก์ชันพื้นฐาน

namespace GradeCheckerApp
{
    class Program
    {
        static void Main(string[] args)
        {
            // ==========================================
            // ส่วนที่ 1: วาด User Interface (UI) เริ่มต้น
            // ==========================================
            Console.WriteLine("=================================================");
            Console.WriteLine("       🎓 UNIVERSITY GRADE SYSTEM V.1.0 🎓       ");
            Console.WriteLine("=================================================");
            
            // รับชื่อนักศึกษา
            Console.Write("กรุณากรอกชื่อนักศึกษา: ");
            string studentName = Console.ReadLine() ?? "ไม่ระบุชื่อ";

            // ==========================================
            // ส่วนที่ 2: ดักจับ Error ระดับแครช (Challenge 1)
            // ==========================================
            Console.Write("กรุณากรอกคะแนนที่ได้ (0-100): ");
            string inputScore = Console.ReadLine();
            
            // สร้างตัวแปรรอรับคะแนนจากการแปลง
            int score;
            
            // ใช้ TryParse เพื่อความปลอดภัย ถ้าผู้ใช้พิมพ์อักษรมั่วมา (เช่น "ร้อย") มันจะเข้า else ทันทีโดยโปรแกรมไม่เด้งหลุด
            if (!int.TryParse(inputScore, out score))
            {
                Console.WriteLine("\n[ กำลังประมวลผลข้อมูล... ]\n");
                Console.WriteLine("❌ CRITICAL ERROR: รูปแบบไม่ถูกต้อง!");
                Console.WriteLine("กรุณากรอกตัวเลขจำนวนเต็มเท่านั้น (ไม่อนุญาตให้ใช้ตัวอักษร)");
                // บังคับจบการทำงานของโปรแกรมตั้งแต่ตรงนี้เลย
                return; 
            }

            Console.WriteLine("\n[ กำลังประมวลผลข้อมูล... ]\n");

            // ==========================================
            // ส่วนที่ 3: ดักจับ Error ขอบเขตตัวเลข (Validation)
            // ==========================================
            if (score < 0 || score > 100)
            {
                Console.WriteLine("❌ ERROR: ข้อมูลคะแนนไม่ถูกต้อง!");
                Console.WriteLine("กรุณาระบุคะแนนในช่วง 0 ถึง 100 เท่านั้น ระบบได้ทำการระงับการออกเกรด");
                return; // จบการทำงาน
            }

            // ==========================================
            // ส่วนที่ 4: ตรรกะการประมวลผลตัดเกรด (Business Logic)
            // ==========================================
            // สร้างตัวแปรรอรับข้อมูล
            string grade = "";
            string statusMsg = "";
            
            // ถ้าหลุดมาถึงตรงนี้ได้ แสดงว่า score ปลอดภัยชัวร์ 100% (เป็นตัวเลข และอยู่ระหว่าง 0-100)
            if (score >= 80)
            {
                grade = "A";
                statusMsg = "🌟 ผ่านเกณฑ์ยอดเยี่ยม!";
            }
            else if (score >= 70)
            {
                grade = "B";
                statusMsg = "ผ่านเกณฑ์ดี";
            }
            else if (score >= 60)
            {
                grade = "C";
                statusMsg = "ผ่านเกณฑ์ปานกลาง";
            }
            else if (score >= 50)
            {
                grade = "D";
                statusMsg = "ผ่านเกณฑ์ขั้นต่ำ";
            }
            else
            {
                grade = "F";
                statusMsg = "💥 ไม่ผ่านเกณฑ์ ต้องลงเรียนใหม่";
            }

            // ==========================================
            // ส่วนที่ 5: การประมวลผล GPA และ ข้อความเตือน (Challenge 2 & 3)
            // ==========================================
            // นำตัวแปร grade ที่ได้จาก if-else ข้างบน มาเข้าเครื่อง Switch Expression ยุคใหม่
            double gpa = grade switch
            {
                "A" => 4.0,
                "B" => 3.0,
                "C" => 2.0,
                "D" => 1.0,
                "F" => 0.0,
                _ => 0.0 // ป้องกัน Error แฝง
            };

            // ใช้ Ternary Operator บรรทัดเดียว เพื่อแยกสถานะปลอดภัย/เสี่ยงรีไทร์ โดยอิงจากเกรดเฉลี่ย
            string survivalStatus = (gpa >= 2.0) ? "(สถานะปลอดภัย)" : "(สถานะโปรเบชั่น/เสี่ยงรีไทร์)";

            // ==========================================
            // ส่วนที่ 6: สร้างรายงานและแสดงผล (Output Reporting)
            // ==========================================
            Console.WriteLine("-------------------------------------------------");
            Console.WriteLine("              📄 OFFICIAL TRANSCRIPT 📄          ");
            Console.WriteLine("-------------------------------------------------");
            Console.WriteLine($"  ชื่อนักศึกษา : {studentName}");
            Console.WriteLine($"  คะแนนรวม   : {score} / 100");
            Console.WriteLine($"  ระดับผลการเรียน : {grade}");
            Console.WriteLine($"  เกรดเฉลี่ย (GPA): {gpa:F1} {survivalStatus}"); // :F1 คือการบังคับแสดงทศนิยม 1 ตำแหน่ง (เช่น 4.0 ไม่ใช่ 4)
            Console.WriteLine($"  สถานะระบบ   : {statusMsg}");
            Console.WriteLine("-------------------------------------------------");
            Console.WriteLine(">> ลงชื่อ ระบบอัตโนมัติ AI Registrar");
        }
    }
}
```

### 🧠 บทวิเคราะห์และข้อสังเกตจากโค้ด (Key Takeaways)

1. **ลำดับขั้นของการป้องกันความผิดพลาด (Defensive Programming)**
   - สังเกตไหมว่าโค้ดมีการป้องกัน 2 ชั้นชัดเจน 
   - ชั้นที่ 1 ป้องกันระดับประเภทข้อมูล: ใช้ `TryParse` ห้ามกรอกตัวหนังสือ ไม่งั้นพัง
   - ชั้นที่ 2 ป้องกันระดับความหมายข้อมูล: ใช้ `if (score < 0 || score > 100)` ห้ามกรอกเลขที่แหกกฎฟิสิกส์ 
   - ทักษะการสร้าง "ด่านตรวจ" เหล่านี้ก่อนที่จะยอมให้ข้อมูลเข้าสู่กระบวนการหลัก (Business Logic) เป็นนิสัยพื้นฐานที่ Software Engineer ระดับโลกทุกคนต้องมี เรียกว่า "Fail-Fast Principle"

2. **พลังของการทำงานร่วมกันระหว่าง เครื่องมือเก่า และ เครื่องมือใหม่**
   - เราใช้เครื่องมือเก่าสุดคลาสสิกอย่าง `if-else if` ในการตัดเกรดจากตัวเลขแบบเป็นช่วงๆ เพราะมันถนัดงานประเภทนี้
   - เราใช้เครื่องมือใหม่สุดเฟี้ยวอย่าง `Switch Expression` ในการแปลงเกรดตัวอักษร ("A") ให้กลายเป็นตัวเลข GPA (4.0) แบบตรงตัวเป๊ะๆ ซึ่งสวยงามและไร้รอยต่อ
   - เราใช้เครื่องมือเฉพาะทางแบบ Ternary `?:` ในการตัดสินใจเรื่องเล็กๆ น้อยๆ เพื่อลดความบวมของโค้ด
