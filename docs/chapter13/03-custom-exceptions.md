# Custom Exceptions (สร้างปีศาจของคุณเอง)

> 💡 **เป้าหมาย:** สัมผัสความรู้สึกของการเป็น "สถาปนิกผู้คุมกฎ"! ในโลก Enterprise โค้ดที่มีคุณภาพจะไม่ได้ทำแค่ดักจับ Error ของชาวบ้าน แต่มันจะ "โยน (Throw)" Error สายพันธุ์ที่ตั้งชื่อขึ้นมาเอง เพื่อบังคับให้ระบบทำงานอยู่ในลู่ทางของ Business Logic อย่างเคร่งครัด!

## 📖 ทฤษฎีและแนวคิดระดับลึก (In-Depth Theory & Concepts)

### 1. ทำไมเราต้องสร้าง Exception เอง?

สมมติคุณเขียนระบบธนาคาร แล้วผู้ใช้พยายามถอนเงิน 10,000 บาท ทั้งๆ ที่มีเงินแค่ 500 บาท
- คุณใช้ `if` เช็คแล้วพบว่าเงินไม่พอ... คำถามคือ **คุณจะทำอย่างไรต่อ?**
- จะ `return false;` หรอ? (แล้วหน้าจอจะรู้ได้ไงว่า false เพราะเงินไม่พอ หรือ false เพราะระบบล่ม?)
- จะ `throw new Exception("เงินไม่พอ");` หรอ? (หน้าจอก็ต้องไปนั่งเอา string มาเปรียบเทียบคำว่า "เงินไม่พอ" ซึ่งโคตรจะเปราะบาง)

**ทางออกระดับโลก:** เราจะสร้างออบเจกต์ระเบิดสายพันธุ์ใหม่ของเราเองเลย! ชื่อว่า **`InsufficientFundsException`** (ระเบิดเงินไม่พอ)
พอหน้าจอเห็นระเบิดลูกนี้พุ่งมา มันจะรู้ตัวทันทีว่า "อ้อ ต้องขึ้นป๊อปอัปให้เติมเงิน" โดยไม่ต้องไปนั่งอ่านข้อความ String เลยแม้แต่น้อย!

### 2. กระบวนการตีเหล็กสร้างระเบิด (Creating a Custom Exception)

การสร้าง Custom Exception ใน C# นั้นง่ายจนน่าตกใจ! 
คุณแค่สร้างคลาสใหม่ แล้วบอกให้มัน **"สืบทอดสายเลือด (Inherit) มาจากคลาสปู่ทวด `Exception`"** เท่านั้นเอง!

```csharp
// 1. สร้างคลาสระเบิดสายพันธุ์ใหม่ (ควรลงท้ายชื่อด้วยคำว่า Exception เสมอ)
public class InsufficientFundsException : Exception
{
    // 2. สร้าง Constructor ที่รับข้อความ แล้วส่งทะลุขึ้นไปหาปู่ทวด (ด้วยคำว่า base)
    public InsufficientFundsException(string message) : base(message)
    {
        // ว่างเปล่าได้เลย ปู่ทวดจัดการให้หมดแล้ว!
    }
}
```

### 3. การโยนระเบิดใส่หน้าผู้เรียก (The `throw` Keyword)

เมื่อคุณเจอกฎของ Business Logic ถูกฝ่าฝืน คุณสามารถเสกระเบิดขึ้นมาแล้วใช้คำสั่ง **`throw`** ปามันใส่หน้าคนที่เรียกใช้ฟังก์ชันคุณได้เลย! (ใครใช้ฟังก์ชันนี้ต้องรับผิดชอบไปกางตาข่าย `try/catch` เอาเอง!)

```csharp
public void Withdraw(int amount) 
{
    if (amount > Balance) {
        // เสกระเบิดสายพันธุ์ตัวเอง แล้วปาออกไป! 
        // 💥 ฟังก์ชันนี้จะหยุดทำงานบรรทัดนี้ทันที! (return อัตโนมัติ)
        throw new InsufficientFundsException("แหกตาดูยอดเงินด้วยครับ!");
    }
    Balance -= amount; // ถ้าไม่มีระเบิด ถึงจะลงมาหักเงินตรงนี้
}
```

## 💻 ตัวอย่างโค้ดเชิงลึก (In-Depth Implementation)

### ตัวอย่าง: สถาปัตยกรรมระบบคัดกรองผับหรู (The Bouncer System)

เราจะจำลองระบบการ์ดคุมหน้าผับ ที่มีความเข้มงวดสุดๆ ถ้าลูกค้าอายุไม่ถึง หรือใส่รองเท้าแตะมา การ์ดจะปาระเบิดไล่ตะเพิดกลับไป (Throw Exceptions)!

::: code-group
```csharp [Program.cs]
using System;

namespace CustomExceptionsConcept
{
    // ==========================================================
    // 1. โรงงานผลิตอาวุธระเบิด (Custom Exceptions)
    // ==========================================================
    
    // ระเบิดสายพันธุ์ที่ 1: อายุไม่ถึงเกณฑ์
    public class UnderageException : Exception
    {
        // แถมลูกเล่น: เก็บอายุที่แท้จริงไว้ประจานด้วย!
        public int ActualAge { get; private set; } 

        public UnderageException(string message, int actualAge) : base(message)
        {
            ActualAge = actualAge; // เก็บหลักฐาน
        }
    }

    // ระเบิดสายพันธุ์ที่ 2: แต่งตัวผิดระเบียบ
    public class DressCodeException : Exception
    {
        public DressCodeException(string message) : base(message) { }
    }

    // ==========================================================
    // 2. โลจิกแกนกลาง (The Bouncer / Business Logic)
    // ==========================================================
    class NightClub
    {
        public void Enter(string name, int age, bool wearSlippers)
        {
            Console.WriteLine($"[การ์ด]: กำลังตรวจร่างกายคุณ {name}...");

            // กฎข้อที่ 1: อายุต้อง 20+
            if (age < 20)
            {
                // โยนระเบิดก้อนที่ 1 พร้อมฝังหลักฐานอายุลงไป
                throw new UnderageException("ไปกินนมแล้วนอนซะไอ้หนู!", age);
            }

            // กฎข้อที่ 2: ห้ามใส่แตะ
            if (wearSlippers)
            {
                // โยนระเบิดก้อนที่ 2
                throw new DressCodeException("ใส่รองเท้าแตะมาผับหรู บ้าไปแล้ว!");
            }

            Console.WriteLine($"[การ์ด]: เชิญด้านในครับคุณ {name} ขอให้สนุก!");
        }
    }

    // ==========================================================
    // 3. ฝั่งผู้ใช้งาน (The Client / UI Level)
    // ==========================================================
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine("=== ระบบสแกนคนเข้าผับ VVIP ===\n");

            NightClub club = new NightClub();

            try 
            {
                // ลองส่งเด็ก 15 ใส่แตะเข้าไป (รับรองโดนเตะก้านคอตั้งแต่ด่านอายุ)
                club.Enter("Arthur", 15, true); 
            }
            // 🌟 ดักจับแบบแยกประเภท เพื่อให้ UI สามารถแสดงผลได้แตกต่างกัน!
            catch (UnderageException ex)
            {
                Console.ForegroundColor = ConsoleColor.Red;
                // สังเกตว่าเราสามารถดึง Property ลับ 'ActualAge' ที่เราซ่อนไว้ในระเบิดออกมาโชว์ได้ด้วย!
                Console.WriteLine($"❌ โดนเด้ง (เรื่องอายุ): {ex.Message} (อายุแค่ {ex.ActualAge} ขวบเอง)");
            }
            catch (DressCodeException ex)
            {
                Console.ForegroundColor = ConsoleColor.Yellow;
                Console.WriteLine($"❌ โดนเด้ง (เรื่องชุด): {ex.Message}");
            }
            catch (Exception ex)
            {
                Console.WriteLine("❌ โดนเด้งด้วยเหตุผลลึกลับ!");
            }

            Console.ResetColor();
        }
    }
}
```
:::

## 🎯 โจทย์ฝึกปฏิบัติเสริมความเข้าใจ (Mini Exercise)

### **โจทย์ที่ 1: การสืบทอดสายเลือดที่ผิดเพี้ยน**
มีเด็กฝึกงานสร้าง Exception ขึ้นมาเองแบบนี้:
```csharp
public class MathErrorException {
    public string Message { get; set; }
    public MathErrorException(string msg) { Message = msg; }
}
```
พอเขาลองสั่ง `throw new MathErrorException("พัง!");` คอมไพเลอร์กลับขีดเส้นแดงปรี๊ดดด ด่าว่าไม่สามารถ Throw ได้! 
**คำถาม:** เขาทำผิดสถาปัตยกรรมระดับชาติข้อไหนครับ? ทำไมคอมพิวเตอร์ถึงไม่ยอมให้ก้อนคลาสนี้ถูกใช้เป็นระเบิด?

### **โจทย์ที่ 2: ระเบิดซ้อนระเบิด (The Base Overlap)**
ถ้าคุณสร้าง `public class MyBugException : Exception { }` 
แล้วคุณไปเขียนด่านสกัดตามลำดับนี้:
```csharp
try { throw new MyBugException(); }
catch (Exception ex) { Console.WriteLine("A"); }
catch (MyBugException ex) { Console.WriteLine("B"); }
```
**คำถาม:** ผลลัพธ์จะปริ้น A หรือ B ครับ? และมันขัดกับหลักการวางด่านสกัดในบทที่แล้วอย่างไร?

::: details 💡 คำใบ้และแนวทาง (Hints)
- **สำหรับโจทย์ 1:** **ลืมประกาศสืบทอด `: Exception` ครับ!**
  ในภาษา C# ไม่ใช่ว่าคุณจะคว้าคลาสหน้ากากแมวที่ไหนมาทำเป็นระเบิดโยนใส่กันได้นะครับ! คีย์เวิร์ด `throw` ยอมรับเฉพาะออบเจกต์ที่มีสายเลือดสืบทอดมาจากคลาส `System.Exception` (ของไมโครซอฟต์) เท่านั้นครับ! ถ้าลืมใส่ `: Exception` ต่อท้ายชื่อคลาส มันก็คือคลาสเก็บ Data โง่ๆ ธรรมดาตัวนึงครับ ปาไม่ได้!
- **สำหรับโจทย์ 2:** **แดงเถือกครับ Compile ไม่ผ่าน! (Compile Error)**
  นี่คือการตอกย้ำเรื่องลำดับชั้นครับ! กฎคือ "ตาข่ายตาถี่ (ลูก) ต้องอยู่บน ตาข่ายตากว้าง (แม่) ต้องอยู่ล่าง" 
  คุณเอา `Exception` ซึ่งเป็นแม่ของทุกสรรพสิ่งไปกางรับเป็นด่านแรก... คอมไพเลอร์จะด่าทันทีว่า "เห้ย! MyBug มันก็เป็นลูกของ Exception นี่หว่า ถ้าเอ็งเอาแม่ขวางไว้ MyBug ด่านสองก็กลายเป็นหมันสิวะ!" 
  คอมไพเลอร์ของ C# ฉลาดมากครับ มันจะดักไม่ให้คุณ Build โค้ดผ่านเลยตั้งแต่แรก! (ต้องสลับเอา MyBug ขึ้นไปไว้บนสุดครับ)
:::

## 🔥 Challenge (โจทย์ท้าทายสุดโหด!)

**โจทย์: การจู่โจมของมังกรไฟ (The Dragon Boss Encounter)**

**ภารกิจของคุณ:** จงสร้างระเบิด 2 ลูก เพื่อจำลองกติกาการสวมใส่อาวุธในเกม RPG!

**ข้อมูลตั้งต้น:** คุณกำลังเขียนฟังก์ชัน `public void EquipWeapon(int playerLevel, string playerClass)` 

**เป้าหมาย:** 
1. สร้าง Custom Exception ชื่อ **`LevelTooLowException`** (สืบทอดจาก Exception)
2. สร้าง Custom Exception ชื่อ **`WrongClassException`** (สืบทอดจาก Exception)
3. ในฟังก์ชัน `EquipWeapon`:
   - ถ้า `playerLevel` น้อยกว่า 50... ให้ Throw **`LevelTooLowException`** พร้อมด่าว่า "เลเวลคุณกากเกินไปสำหรับดาบมังกร!"
   - ถ้า `playerClass` ไม่ใช่ "Warrior"... ให้ Throw **`WrongClassException`** พร้อมด่าว่า "นักเวทย์ห้ามถือดาบเฟ้ย!"
   - ถ้าผ่าน 2 ด่านนี้ได้ ให้ปริ้น "สวมใส่ดาบมังกรสำเร็จ พลังโจมตี +999!"
4. ใน `Main` ลองเรียก `EquipWeapon(10, "Mage")` โดยเอา `try/catch` ครอบไว้ 2 ชั้น เพื่อรับมือกับ Exception ทั้ง 2 สายพันธุ์ และโชว์ข้อความให้สวยงาม!

นี่คือสถาปัตยกรรมแบบ **"Fail-Fast (ให้พังเร็วที่สุด)"** ที่วงการซอฟต์แวร์ใช้กันครับ! คือถ้าเงื่อนไขไม่ผ่าน เราไม่ปล่อยให้มันเดินลึกเข้าไปในโค้ดหรอก เราปาระเบิดไล่ตะเพิดตั้งแต่ 2 บรรทัดแรกเลย! โค้ดจะสะอาดมาก!

## 🗣️ ทบทวน (Review)

::: details ❓ คำถามทบทวนความเข้าใจเชิงลึก

**คำถาม 1:** การที่เรามัวแต่ใช้ `throw new ...` ปาระเบิดกันไปมาแบบนี้ มันกิน CPU (Performance) หนักกว่าการใช้ `if` แล้วค่อย `return false` ไหมครับ?
**แนวคำตอบ:** **หนักกว่ามหาศาลครับ! (โคตรจะหนักเลย)**
นี่คือเรื่องจริงในระดับวิศวกรรมซอฟต์แวร์ครับ: การสร้าง (Instantiate) Exception 1 ลูก แล้วปามันขึ้นฟ้า... คอมพิวเตอร์ต้องทำการจำลองสภาพแวดล้อม (Stack Trace) ถ่ายรูปเก็บว่าพังที่บรรทัดไหน ไฟล์ไหน ฟังก์ชันไหนเรียกมา... ซึ่งใช้ CPU หนักกว่าการ `return false` เป็นหมื่นเท่า!
**ข้อสรุปของสถาปนิก:** เราจึงควรปาระเบิด Exception เฉพาะเหตุการณ์ที่ "ผิดปกติแบบหาทางออกไม่ได้จริงๆ (Exceptional Flow)" เท่านั้นครับ! ถ้าเป็นลอจิกปกติ เช่น ล็อกอินรหัสผ่านผิด... สถาปนิกจะให้ชดใช้ `return false` เป็นหลักครับ ไม่ควรใช้ Exception มาควบคุม Flow ของโปรแกรมปกติเด็ดขาด! (Anti-Pattern: Control Flow via Exceptions)

**คำถาม 2:** ผมสามารถ Throw ทิ้งไว้ดื้อๆ ได้ไหมครับ โดยที่ปลายทางใน Main ไม่มีใครเขียน `catch` ดักมันเอาไว้เลย?
**แนวคำตอบ:** **ทำได้ครับ... แต่วินาทีที่มันทำงาน โปรแกรมคุณจะปิดตัวกระจุย (Unhandled Exception Crash) ทันทีครับ!**
มันเหมือนคุณดึงสลักระเบิดมือในห้องประชุม แล้วไม่มีการ์ดวิ่งเอาผ้าห่มมากางทับ (catch) ครับ! 
ระเบิดมันจะทำงานของมัน 100% พุ่งกระจายขึ้นไปทะลุเพดาน จนไปถึงระบบ Windows (OS)... Windows เห็นว่าโปรแกรมนี้คุมตัวเองไม่ได้ ก็จะสั่งดับกระบวนการ (Terminate Process) ทิ้ง เพื่อไม่ให้รามไปติดส่วนอื่นของคอมพิวเตอร์ครับ! ดังนั้น ถ้าคุณจะโยนระเบิด คุณต้องมั่นใจว่าฝั่ง UI หน้าจอของคุณ มีตาข่ายนิรภัยกางรอมันอยู่เสมอครับ!
:::
