# เฉลย: Custom Exceptions (สร้างปีศาจของคุณเอง)

## 🎯 เฉลย Mini Exercise

### **โจทย์ที่ 1: การสืบทอดสายเลือดที่ผิดเพี้ยน**
**เป้าหมาย:** ชี้แจงสิทธิ์ในการถือครองพลัง "ระเบิด (throw)" ของระบบ OOP

**เฉลยคำตอบระดับสถาปนิก:**
เขาทำผิดกฎเหล็กเรื่อง **"สายเลือดมรดก (Inheritance)"** ครับ!
ใน C# คีย์เวิร์ด `throw` มันมีเงื่อนไขศักดิ์สิทธิ์อยู่ว่า: *"ข้าจะยอมขว้างเฉพาะออบเจกต์ ที่สืบเชื้อสาย (Inherit) มาจากตระกูล `System.Exception` เท่านั้น!"*
เมื่อเด็กฝึกงานลืมใส่โคลอน `: Exception` ต่อท้ายชื่อคลาส... คลาสนั้นจึงเป็นแค่กล่องเก็บข้อมูลโง่ๆ ตัวนึง (เหมือนกล่อง Student หรือ Employee ทั่วไป) 
คอมไพเลอร์เห็นปุ๊บก็เลยด่าเปิงเลยครับว่า "เห้ย นี่มันกล่องเปล่าๆ เอ็งจะเอามาขว้างเป็นระเบิดได้ยังไงวะ! ไปต่อเติมสายเลือดมาให้เรียบร้อยก่อน!"

### **โจทย์ที่ 2: ระเบิดซ้อนระเบิด (The Base Overlap)**
**เป้าหมาย:** ย้ำเตือนกฎเหล็กเรื่องลำดับการดักจับ (Catch Order) อีกครั้งแบบพลิกแพลง

**เฉลย:**
ผลลัพธ์คือ **"แดงเถือก Compile ไม่ผ่านครับ!"**
มันขัดกับหลักการ "ลูกขวางแม่ไม่ได้" อย่างรุนแรงครับ 
คุณเอา `Exception` (แม่ใหญ่) ไปกางตาข่ายไว้ด่านบนสุด... พอมันมีคนโยนระเบิด `MyBugException` มา... แม่ใหญ่มันก็จะอ้าแขนรับไว้เองหมดเกลี้ยงเลยครับ เพราะตามหลัก Polymorphism (บท 10) "MyBug ถือว่าเป็น Exception ตัวนึง"! 
ทำให้บล็อก catch ด่านล่างสุด (ที่รอรับ MyBug โดยเฉพาะ) กลายเป็น **"พื้นที่ที่ไม่มีวันไปถึงตลอดกาล (Unreachable Code)"** ซึ่งคอมไพเลอร์ C# มองว่าเป็นบั๊กร้ายแรง และสั่งห้ามไม่ให้คุณ Build โปรแกรมผ่านเด็ดขาดครับ!

## 🔥 เฉลย Challenge (การจู่โจมของมังกรไฟ The Dragon Boss Encounter)

**เป้าหมาย:** สัมผัสความสุดยอดของ **Fail-Fast Design!** นี่คือสถาปัตยกรรมแบบเดียวกับที่ใช้ในระบบ Backend Web API ระดับโลก! โค้ดในส่วนประมวลผล (Business Logic) จะบางเฉียบ สะอาดสะอ้าน เพราะหน้าที่ด่าผู้ใช้ทั้งหมด ถูกโยนเป็นระเบิดส่งกลับไปให้หน้าจอ (UI) จัดการแทน!

**กระบวนการคิดระดับสถาปนิก (Fail-Fast Throwing):**
เราจะสร้าง Custom Exception 2 ตัว แล้วเอาไปแปะเป็นกฎคัดกรองตั้งแต่บรรทัดแรกๆ ของฟังก์ชัน ถ้าผู้เล่นไม่ผ่านเกณฑ์ ถีบหัวส่งปาระเบิดกลับไปทันที ไม่ต้องให้เดินลึกเข้าไปข้างใน!

```csharp
using System;

namespace DragonBossChallenge
{
    // ==========================================================
    // 1. โรงงานผลิตระเบิดคู่ (The Custom Exceptions)
    // สืบทอดพลังทำลายล้างมาจากปู่ทวด Exception ทั้งคู่
    // ==========================================================
    public class LevelTooLowException : Exception
    {
        public LevelTooLowException(string message) : base(message) { }
    }

    public class WrongClassException : Exception
    {
        public WrongClassException(string message) : base(message) { }
    }

    class Program
    {
        // ==========================================================
        // 2. ลอจิกเกม (Business Logic) -> สถาปัตยกรรม Fail-Fast
        // ==========================================================
        public static void EquipWeapon(int playerLevel, string playerClass)
        {
            Console.WriteLine($"[ระบบ]: ตรวจสอบคุณสมบัติ ผู้เล่นเลเวล {playerLevel} อาชีพ {playerClass}...");

            // กฎข้อ 1: เลเวลต้อง 50 ขึ้นไป
            if (playerLevel < 50) {
                // ปาระเบิดทิ้ง แล้วหยุดการทำงานฟังก์ชันนี้ทันที!
                throw new LevelTooLowException("เลเวลคุณกากเกินไปสำหรับดาบมังกร!");
            }

            // กฎข้อ 2: ต้องเป็น Warrior เท่านั้น
            if (playerClass != "Warrior") {
                // ปาระเบิดลูกที่สองทิ้ง!
                throw new WrongClassException("นักเวทย์ห้ามถือดาบเฟ้ย! ไปถือไม้กายสิทธิ์นู่น!");
            }

            // ถ้าผ่านมาถึงตรงนี้ได้ แปลว่ารอด 100% ไม่มีบั๊กแอบแฝง
            Console.ForegroundColor = ConsoleColor.Green;
            Console.WriteLine("✨ สวมใส่ดาบมังกรสำเร็จ พลังโจมตี +999!");
            Console.ResetColor();
        }

        // ==========================================================
        // 3. หน้าจอเกม (The Client UI Level)
        // ==========================================================
        static void Main(string[] args)
        {
            Console.WriteLine("=== ระบบกระเป๋าสัมภาระ (Inventory) ===\n");

            try
            {
                // ลองของ! ส่งนักเวทย์เลเวล 10 ไปถือดาบเทพ!
                EquipWeapon(10, "Mage"); 
                
                // ถ้ารอด มันจะปริ้นบรรทัดนี้ แต่ไม่มีทางรอดหรอกฮ่าๆ
                Console.WriteLine("เดินหน้าลุยบอส!"); 
            }
            // 🌟 หน้าจอมีหน้าที่แค่ "กางตาข่าย" รอรับผลกรรมที่ฟังก์ชันข้างบนโยนกลับมา
            catch (LevelTooLowException ex)
            {
                Console.ForegroundColor = ConsoleColor.Yellow;
                Console.WriteLine($"⚠️ [ระบบปฏิเสธ]: {ex.Message}");
            }
            catch (WrongClassException ex)
            {
                Console.ForegroundColor = ConsoleColor.Red;
                Console.WriteLine($"❌ [ระบบปฏิเสธ]: {ex.Message}");
            }
            catch (Exception ex)
            {
                // ดักกันเหนียว เผื่อระบบพังลึกกว่านั้น
                Console.WriteLine($"[วิกฤติ]: พังยับ {ex.Message}");
            }

            Console.ResetColor();

            /*
               ผลลัพธ์ที่ได้:
               [ระบบ]: ตรวจสอบคุณสมบัติ ผู้เล่นเลเวล 10 อาชีพ Mage...
               ⚠️ [ระบบปฏิเสธ]: เลเวลคุณกากเกินไปสำหรับดาบมังกร!  <-- โดนถีบกระเด็นตั้งแต่ด่านแรก (ด่านเลเวล)
            */
        }
    }
}
```

**สิ่งที่ได้เรียนรู้ (Architecture Lesson):**
สถาปัตยกรรมแบบ **"โยน (Throw) หลังบ้าน -> ดักจับ (Catch) หน้าบ้าน"** คือมาตรฐานระดับทองคำของการสร้าง Software บนโลกใบนี้ครับ!
หน้าที่ของการคิดเลข/ตรวจสอบสิทธิ์ ควรอยู่ในคลาสลอจิก (หลังบ้าน) ส่วนหน้าที่ปริ้นข้อความสีๆ ด่ากราดผู้ใช้ด้วยภาษาคน ควรอยู่ในหน้าจอ Console (หน้าบ้าน) เราใช้ Custom Exception เป็น "จดหมายส่งข่าว" ข้ามมิติหากันระหว่างสองระบบนี้ครับ! โคตรทรงพลัง!
