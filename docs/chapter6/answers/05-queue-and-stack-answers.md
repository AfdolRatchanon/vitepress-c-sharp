## 🎯 เฉลย Mini Exercise

### **โจทย์ที่ 1: เถียงกับกฎระเบียบ**
**คำถาม:** มี `Queue<int>` ที่ใส่คิวเลข 10, 20, 30 ไว้ มีวิธีไหนที่จะดึง 20 ออกมาใช้งาน โดยที่ปล่อยเลข 10 ให้นั่งรออยู่ในคิวเหมือนเดิมไหม?

**เฉลยทางวิศวกรรม:**
**เป็นไปไม่ได้เด็ดขาด 1,000,000% ครับ!**
โครงสร้างข้อมูล Queue ถือกำเนิดขึ้นมาเพื่อ "ตัดสิทธิ์" การกระทำแบบนี้โดยเฉพาะ (ถ้าคุณทำแบบนั้นได้ มันก็จะไม่เรียกว่า Queue แต่มันคือ List ธรรมดา) 
กฎของ Queue คือท่อดำมืดที่มีแค่ทางเข้าท้ายซอย และทางออกหน้าปากซอย ไม่มีประตูด้านข้างให้คุณเอื้อมมือไปล้วงของ 
ดังนั้น ทางเดียวที่คุณจะจับเลข 20 ได้คือ **คุณต้องระเบิดหัวคนแรกทิ้ง (Dequeue เลข 10 ทิ้งไป)** ให้คิวขยับเดินหน้า แล้วคุณถึงจะมีสิทธิ์ได้คุยกับเลข 20 ครับ

### **โจทย์ที่ 2: รหัสมอสกลับหลัง (The Palindrome Backwards)**
**เป้าหมาย:** โยนคำว่า HELLO เข้าไปใน Stack แล้วไล่ดึงออกมาจนหมดเกลี้ยง จะได้คำว่าอะไร?

**เฉลยและวิธีทำ:**
LIFO (Last In, First Out) คือคอนเซปต์เวทมนตร์ของการ "สลับหน้าเป็นหลัง" ที่ทรงพลังที่สุด

```csharp
using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        string word = "HELLO";
        
        // สร้างกระบอกใส่ลูกแบด
        Stack<char> magicStack = new Stack<char>();

        // 1. กด Push เข้าไปทีละตัว
        foreach (char c in word)
        {
            magicStack.Push(c);
            Console.WriteLine($"[ยัดของ] กดอักษร '{c}' ลงสแต็ก");
        }

        // ตอนนี้ H จะอยู่ก้นบ่อลึกสุด และ O จะอยู่ชั้นบนสุด

        Console.Write("\n[ผลลัพธ์ดึงของ] : ");

        // 2. ดึง Pop ออกมาทีละตัว จนกว่ากระบอกจะเกลี้ยง
        while (magicStack.Count > 0)
        {
            // หยิบคนบนสุดหลุดออกมา
            char topChar = magicStack.Pop(); 
            Console.Write(topChar);
        }
        Console.WriteLine();
        // ผลลัพธ์: OLLEH
    }
}
```

## 🔥 เฉลย Challenge (เครื่องตรวจสอบความสมดุลของวงเล็บ The Parentheses Balancer)

**เป้าหมาย:** สร้างอัลกอริทึมในการตรวจสอบว่าวงเล็บ เปิด `(` และ ปิด `)` ของโค้ดเรานั้นสมดุลกันหรือไม่?

**กระบวนการคิดระดับสถาปนิก (Whiteboard Algorithm):**
โจทย์ข้อนี้คือโจทย์สัมภาษณ์งานระดับตำนาน (LeetCode) ที่บริษัท FAANG (Facebook, Apple, Amazon, Netflix, Google) ชอบเอามาใช้คัดกรองโปรแกรมเมอร์

ทำไมเราถึงใช้การ "นับเลข (+1, -1) ธรรมดาๆ" ไม่ได้? ทำไมต้องใช้ Stack?
เพราะในชีวิตจริง วงเล็บมันมีหลายรูปทรง (เช่น `{ [ ( ) ] }`) การใช้ Stack จะช่วยจับคู่ความถูกต้องให้เราได้อย่างแม่นยำที่สุด แต่ในโจทย์นี้เราจะโฟกัสแค่วงเล็บโค้ง `()` อย่างเดียวก่อนเพื่อความง่าย

```csharp
using System;
using System.Collections.Generic;

namespace ParenthesesBalancer
{
    class Program
    {
        // สร้างเมธอดฉลาดๆ คืนค่า true (สมดุล) หรือ false (พัง)
        static bool CheckBalance(string text)
        {
            // 1. สร้างกระบอกสูญญากาศ
            Stack<char> parenthesisStack = new Stack<char>();

            // 2. กวาดสายตาอ่านตัวอักษรไปทีละตัว
            foreach (char c in text)
            {
                if (c == '(')
                {
                    // ----------------------------------------
                    // กฎข้อที่ 1: เจอเปิด -> กดทับลงไป (Push)
                    // ----------------------------------------
                    parenthesisStack.Push(c);
                }
                else if (c == ')')
                {
                    // ----------------------------------------
                    // กฎข้อที่ 2: เจอวงเล็บปิด -> ต้องเคลียร์บิล!
                    // ----------------------------------------
                    
                    // ปัญหา: ถ้ากระบอกว่างเปล่า แปลว่ามีคนแอบปิดวงเล็บ ทั้งที่ไม่ได้เปิด!
                    // (เช่นประโยค "())( " ไอ้ตัวที่ 3 จะทำให้พังตรงนี้แหละ)
                    if (parenthesisStack.Count == 0)
                    {
                        return false; // สั่งฟันธงว่า "ไม่สมดุล" และเตะออกจากเมธอดทันที!
                    }

                    // ถ้ารอดมาได้ แปลว่าในกระบอกยังมีวงเล็บเปิดรอคู่ของมันอยู่
                    // เราก็จัดการ ดึง (Pop) มันออกมาฆ่ากันทิ้งไป (ถือว่าครบคู่แล้ว)
                    parenthesisStack.Pop();
                }
                // ถ้าเป็นตัวอักษรอื่น (a, b, c) เราก็ปล่อยผ่าน ไม่ต้องสนใจมัน
            }

            // 3. กวาดสายตาจนสุดประโยคแล้ว เช็คสภาพกระบอกเป็นครั้งสุดท้าย!
            // ถ้าสมดุลจริง กระบอกต้องไม่มีวงเล็บเปิดตกค้างอยู่เลย (Count == 0)
            if (parenthesisStack.Count == 0)
            {
                return true;  // สมบูรณ์แบบ
            }
            else
            {
                return false; // แปลว่ามีวงเล็บเปิดค้างอยู่แต่ลืมปิด! (เช่น "(hello")
            }
        }

        static void Main(string[] args)
        {
            Console.WriteLine("=== เครื่องตรวจจับวงเล็บ (Syntax Checker) ===\n");

            string[] testCases = {
                "(a + b) * (c - d)",    // สมดุลชัวร์ๆ
                "((x + y)",             // ลืมปิด!
                ")()(",                 // ปิดก่อนเปิด (บ้าไปแล้ว)
                "()(()())",             // ซ้อนกันวุ่นวาย แต่สมดุลนะ
                "Hello World"           // ไม่มีวงเล็บเลย ถือว่าสมดุลในตัวเอง
            };

            foreach (string test in testCases)
            {
                bool isBalanced = CheckBalance(test);
                
                if (isBalanced)
                {
                    Console.ForegroundColor = ConsoleColor.Green;
                    Console.WriteLine($"[PASS] ข้อความสมดุล   => {test}");
                }
                else
                {
                    Console.ForegroundColor = ConsoleColor.Red;
                    Console.WriteLine($"[FAIL] วงเล็บพินาศ! => {test}");
                }
                Console.ResetColor();
            }
        }
    }
}
```
