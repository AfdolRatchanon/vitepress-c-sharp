# เฉลย Lab: Health Data Collector

## 📝 เฉลยฉบับสมบูรณ์ (ครอบคลุมภารกิจพื้นฐานและภารกิจท้าทายสุดโหด)

นี่คือโครงสร้างของโปรแกรมเก็บข้อมูลสุขภาพที่ได้รับการปรับปรุงทางสถาปัตยกรรม (Refactored) เพื่อให้ป้องกันข้อผิดพลาด 100% (Bulletproof) และรวมการประมวลผล BMR เข้าไว้ด้วยกัน 

กรุณาศึกษาเทคนิคการใช้ `double.TryParse` และการจัดการสูตรคณิตศาสตร์ในบล็อก [ส่วนที่ 3] อย่างละเอียด เพราะนั่นคือหัวใจของการเขียนซอฟต์แวร์ประมวลผลข้อมูล (Data Processing) 

```csharp
using System;

namespace HealthDataCollector
{
    class Program
    {
        static void Main(string[] args)
        {
            // ==========================================
            // ส่วนที่ 1: วาดส่วนหัวของโปรแกรม (UI Header)
            // ==========================================
            Console.WriteLine("===================================================");
            Console.WriteLine("     🏥 HOSPITAL HEALTH DATA COLLECTOR V.2 🏥     ");
            Console.WriteLine("               (Advanced Algorithm)                ");
            Console.WriteLine("===================================================");
            Console.WriteLine(); // เว้นบรรทัดเปล่า

            // ==========================================
            // ส่วนที่ 2: ระบบสัมภาษณ์ที่ปลอดภัยจากข้อผิดพลาด (Bulletproof Inputs)
            // ==========================================
            Console.WriteLine("[ระบุข้อมูลประจำตัวคนไข้]");
            
            // รับชื่อ (string ปกติ ไม่ต้องแปลง ไม่มีปัญหาการแครช)
            Console.Write("กรุณากรอกชื่อ-นามสกุล: ");
            string patientName = Console.ReadLine() ?? "ไม่ระบุชื่อ";

            // รับอายุ โดยใช้ TryParse ตามที่ท้าทายไว้ (ถ้ากรอกมั่ว จะกลายเป็น 0 อัตโนมัติ)
            Console.Write("กรุณากรอกอายุ (ปี): ");
            string ageInput = Console.ReadLine();
            int age;
            int.TryParse(ageInput, out age); // ถ้ากรอกอักษร age จะเป็น 0 และโปรแกรมไม่แครช

            Console.WriteLine("\n[ระบุข้อมูลสรีระร่างกาย]");

            // รับน้ำหนักด้วย TryParse แบบ double
            Console.Write("กรุณากรอกน้ำหนักตัว (กิโลกรัม): ");
            string weightInput = Console.ReadLine();
            double weightKg;
            double.TryParse(weightInput, out weightKg);

            // รับส่วนสูง (เซนติเมตร) ด้วย TryParse
            Console.Write("กรุณากรอกส่วนสูง (เซนติเมตร): ");
            string heightInput = Console.ReadLine();
            double heightCm;
            double.TryParse(heightInput, out heightCm);

            Console.WriteLine("\n[ กำลังประมวลผลข้อมูลทางคณิตศาสตร์... ]\n");

            // ==========================================
            // ส่วนที่ 3: อัลกอริทึมการคำนวณทางการแพทย์ (Medical Algorithms)
            // ==========================================
            
            // 3.1 การเตรียมข้อมูล (Data Preparation)
            // ต้องแปลงส่วนสูงจาก เซนติเมตร ให้เป็น เมตร (หาร 100) สำหรับสูตร BMI
            double heightMeters = heightCm / 100.0;

            // 3.2 คำนวณค่า BMI (ดัชนีมวลกาย)
            // สูตร: BMI = น้ำหนัก(kg) / ส่วนสูง(m)^2
            // ข้อควรระวัง: เราป้องกันการหารด้วยศูนย์ (Divide by Zero) เผื่อในกรณีที่ผู้ใช้กรอกส่วนสูงมาเป็น 0
            double bmi = 0;
            if (heightMeters > 0)
            {
                bmi = weightKg / (heightMeters * heightMeters);
            }

            // 3.3 คำนวณค่า BMR (อัตราการเผาผลาญพื้นฐาน) - สำหรับผู้ชาย
            // สูตร: 66 + (13.7 x น้ำหนัก) + (5 x ส่วนสูงcm) - (6.8 x อายุ)
            double bmr = 66.0 + (13.7 * weightKg) + (5.0 * heightCm) - (6.8 * age);

            // ==========================================
            // ส่วนที่ 4: ระบบออกรายงาน (Report Generation)
            // ==========================================
            
            // สร้างรหัสคนไข้ปลอมๆ
            string patientId = $"PATIENT-{age}";

            Console.WriteLine("---------------------------------------------------");
            Console.WriteLine("             📋 สรุปรายงานสุขภาพคนไข้ 📋             ");
            Console.WriteLine("---------------------------------------------------");
            
            // นำเสนอข้อมูลด้วย String Interpolation
            Console.WriteLine($"รหัสอ้างอิง        : {patientId}");
            Console.WriteLine($"ชื่อผู้ป่วย        : {patientName}");
            Console.WriteLine($"อายุ            : {age} ปี");
            
            // แสดงผลลัพธ์พร้อมบังคับทศนิยม 2 ตำแหน่งด้วย :F2
            Console.WriteLine($"ดัชนีมวลกาย (BMI) : {bmi:F2}");
            Console.WriteLine($"การเผาผลาญ (BMR) : {bmr:N0} กิโลแคลอรี่/วัน"); // :N0 ใส่ลูกน้ำและปัดเศษทิ้ง
            
            Console.WriteLine("---------------------------------------------------");
            Console.WriteLine("ข้อแนะนำทางการแพทย์: กรุณานำค่า BMI เทียบกับตารางเกณฑ์มาตรฐาน");
            Console.WriteLine("ระบบอัตโนมัติทำงานเสร็จสมบูรณ์.");
        }
    }
}
```

### 🧠 บทวิเคราะห์ข้อควรระวัง (Key Takeaways & Lessons Learned)

1. **ปัญหาหารด้วยศูนย์ (Divide by Zero Exception):** 
   - ในขั้นตอนที่ 3.2 ผมได้เพิ่มการเช็ค `if (heightMeters > 0)` เอาไว้ เพราะถ้าเราใช้ `TryParse` แล้วพยาบาลพิมพ์ส่วนสูงผิดเป็นอักษร ตัวแปร `heightCm` จะกลายเป็นเลข `0` ทันที 
   - เมื่อคอมพิวเตอร์พยายามคำนวณ `น้ำหนัก / (0 * 0)` ในโลกคณิตศาสตร์การหารด้วยศูนย์คือสิ่งที่เป็นไปไม่ได้! และมันจะทำให้ค่า BMI กลายเป็นสถานะอินฟินิตี้ (Infinity) หรือแย่สุดคือระเบิด Crash การเขียนดักไว้ก่อนจึงเป็นความรอบคอบระดับโปรเฟสชันนอล!
2. **การทำ Type Promotion ในคณิตศาสตร์:** 
   - ตอนหาร 100 สังเกตไหมว่าผมเขียน `100.0` แทนที่จะเป็น `100` เฉยๆ 
   - แม้ `heightCm` จะเป็น `double` อยู่แล้ว การหาร 100 เฉยๆ ก็ใช้งานได้ แต่นิสัยการใส่ทศนิยมกำกับไว้ (เช่น 100.0 หรือ 100d) เป็นการป้องกันไม่ให้ C# เผลอมองมันเป็น Integer Division (ตัดเศษทิ้ง) ในเคสที่ตัวตั้งดันเป็น `int` ด้วยนั่นเอง
3. **การจัดการการแสดงผล:** 
   - ค่า BMR คือพลังงานแคลอรี่ที่ร่างกายต้องการต่อวัน เช่น `1750 Kcal` มันไม่จำเป็นต้องมีทศนิยม เราจึงใช้ `{bmr:N0}` เพื่อปัดทศนิยมทิ้งและใส่ลูกน้ำหลักพันให้ดูเป็นระเบียบสวยงาม
